```c++
#include <iostream>
#include <vector>

using namespace std;

int main(int argc, char *argv[])
{
    int n,temp;
    vector<int> v1;
    vector<int> v2;
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>temp;
        v1.push_back(temp);
    }
    v2.push_back((v1[0]+v1[1])/2);
    for(int i=1;i<n-1;i++){
        v2.push_back((v1[i]+v1[i+1]+v1[i-1])/3);
    }
    v2.push_back((v1[n-1]+v1[n-2])/2);
    for(int i=0;i<n;i++){
        cout<<v2[i]<<" ";
    }
    return 0;
}
```
