```c++
#include <iostream>

using namespace std;


int main()
{
    int s[100],n,num;
    cin>>n;
    for(int i=0;i<100;i++){
        s[i]=i+1;
    }
    for(int i=0;i<n;i++){
        cin>>num;
        //遍历20排
        for(int row=0;row<20;row++){
            int sum=0;
            //统计每排的空余座位数量
            for(int col=0;col<5;col++){
                if(s[row*5+col]!=-1){
                    sum++;
                }
            }
            //如果这排满足数量要求  购买这排  跳出循环
            if(sum>=num){
                for(int col=0;col<5&&num>0;col++){
                    if(s[row*5+col]!=-1){
                        cout<<s[row*5+col]<<" ";
                        s[row*5+col]=-1;
                        num--;
                    }
                }
                break;
            }
        }
        //num不为零 说明没有满足整排的  进行零散购票
        if(num>0){
            for(int j=0;j<100&&num>0;j++){
                if(s[j]!=-1){
                    cout<<s[j]<<" ";
                    s[j]=-1;
                    num--;
                }
            }
        }
        cout<<endl;


    }
    return 0;
}
```
