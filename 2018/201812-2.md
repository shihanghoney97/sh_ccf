```c++
#include <iostream>

using namespace std;

int main()
{
    int r,y,g,n,a,b;
    long long sum=0;
    cin>>r>>y>>g;
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>a>>b;
        int temp;
        switch (a) {
        case 0:
            sum+=b;
            break;
        case 1:
            if(sum<b){
                sum+=(b-sum);
                break;
            }
            temp=(sum-b)%(r+y+g);
            if(!(temp>=0&&temp<g)){
                sum+=g+y-temp+r;
            }
            break;
        case 3:
            if(sum<b){
                break;
            }
            temp=(sum-b)%(r+y+g);
            if(temp>=0&&temp<r+y){
                sum+=r+y-temp;
            }
            break;
        case 2:
            if(sum<b){
                sum+=(b-sum+r);
                break;
            }
            temp=(sum-b)%(r+y+g);
            if(temp>=0&&temp<r){
                sum+=r-temp;
            }
            if(temp>=r+g&&temp<r+g+y){
                sum+=r+((r+y+g)-temp);
            }
            break;

        default:
            break;
        }
    }
    cout<<sum<<endl;
    return 0;
}
```
