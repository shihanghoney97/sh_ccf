```c++
#include <iostream>

using namespace std;

int main()
{
    int n,max=-1;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++){
        cin>>a[i];
    }
    for(int i=0;i<n;i++){
        int suml=0,sumr=0;
        for(int j=0;j<i;j++){
            if(a[j]>=a[i]){
                suml+=a[i];
            }else {
                suml=0;
            }
        }
        for(int j=i;j<n;j++){
            if(a[j]>=a[i]){
                sumr+=a[i];
            }else {
                break;
            }
        }
        max=max>suml+sumr?max:suml+sumr;
    }
    cout<<max<<endl;
    return 0;
}
```
