```c++
#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main(int argc, char *argv[])
{
    vector<int> v;
    int size=0,temp=0;
    int min,max;
    cin>>size;
    for(int i=0;i<size;i++){
        cin>>temp;
        v.push_back(temp);
    }
    min=v[0];
    max=v[size-1];
    if(min>max){
        max=min-max;
        min=min-max;
        max=max+min;
    }
    if(size%2==0){
        int l=v[size/2-1],r=v[size/2];
        int sum=l+r;
        if(sum%2==0){
            int mid=sum/2;
            printf("%d %d %d",max,mid,min);
        }else {
            double mid=double(sum)/2;
            printf("%d %.1f %d",max,mid,min);
        }

    }else {
        int mid=v[(size-1)/2];
        cout<<max<<" "<<mid<<" "<<min;
    }
    return 0;
}
```
