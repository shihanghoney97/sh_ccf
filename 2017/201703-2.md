```c++
#include <iostream>
#include <vector>

using namespace std;

//#define DEBUG

int main()
{
    int n,m,b,c;
    cin>>n>>m;
    vector<int> a;
    for(int i=0;i<n;i++){
        a.push_back(i+1);
    }
    for(int i=0;i<m;i++){
        cin>>b>>c;
        for(int j=0;j<n;j++){
            if(a[j]==b){
                int temp=a[j];
                a.erase(a.begin()+j);
                a.insert(a.begin()+j+c,temp);
                break;
            }
        }

#ifdef DEBUG
        for(int i=0;i<n;i++)
            cout<<a[i]<<" ";
        cout<<endl;
#endif
    }
    for(int i=0;i<n;i++)
        cout<<a[i]<<" ";
    cout<<endl;
    return 0;
}
```
