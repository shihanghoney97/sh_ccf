```c++
#include <iostream>
#include <queue>
using namespace std;

int main()
{
    int n,k,head;
    cin>>n>>k;
    int num=0;
    queue<int> q;
    for(int i=1;i<=n;i++){
        q.push(i);
    }
    while(!q.empty()){
        head=q.front();
        q.pop();
        num++;
        if(!(num%k==0||num%10==k)){
            q.push(head);
        }
    }
    cout<<head<<endl;
    return 0;
}
```
