```c++
#include <iostream>
#include <vector>
#include <algorithm>

//#define DEBUG

using namespace std;

typedef struct teacher{
    int w;
    int s;
    int c;
    int e;
}t;

int main()
{
    int N,K;
    cin>>N>>K;
    int key[N];
    for(int i=0;i<N;i++){
        key[i]=i+1;
    }
    t a[K];
    int min=999999,max=-1;
    for(int i=0;i<K;i++){
        cin>>a[i].w>>a[i].s>>a[i].c;
        a[i].e=a[i].s+a[i].c;
        min=min>a[i].s?a[i].s:min;
        max=max<a[i].e?a[i].e:max;
    }
    //i 时间线中的时间点
    for(int i=min;i<=max;i++){
#ifdef DEBUG
        cout<<"时刻 "<<i<<endl;
#endif
        //还钥匙
        vector<int> retu;
        for(int j=0;j<K;j++){
            if(a[j].e==i){
                //记录这个时间点要归还的钥匙
                retu.push_back(a[j].w);
            }
        }

        sort(retu.begin(),retu.end());
        //遍历需要归还的钥匙
        int num=0;
        while (num<retu.size()) {
#ifdef DEBUG
            cout<<"待归还钥匙："<<retu[0]<<endl;
#endif
            //遍历查询钥匙盒的空缺
            for(int k=0;k<N;k++){
                if(key[k]==-1){
                    key[k]=retu[num];
                    num++;
                    break;
                }
            }
        }

        //借钥匙
        for(int j=0;j<K;j++){
            if(a[j].s==i){
                for(int k=0;k<N;k++){
                    if(key[k]==a[j].w){
                        key[k]=-1;
                    }
                }
            }
        }
#ifdef DEBUG
        for(int j=0;j<N;j++){
            cout<<key[j]<<" ";
        }cout<<endl;
#endif
    }
    for(int j=0;j<N;j++){
        cout<<key[j]<<" ";
    }cout<<endl;
    return 0;
}
```
